### 类

- public - 公有权限
- private - 私有权限
- protected - 受保护的权限
- friend - 友元

```cpp
#include <iostream>
using namespace std;

class node {
    private:
        int x, y;

    public:
        void change_x(int val) {
            this->x = val;
        }
        void change_y(int val) {
            this->y = val;
        }
        void show_x() {
            cout << this->x << endl;
        }
        void show_y() {
            cout << this->y << endl;
        }
        void hello();
    
    friend void my_friend(node &t);
};

void node::hello() {
    cout << "hello" << endl;
}

void my_friend(node &t) {
    cout << "friend" << endl;
    cout << t.x << " " << t.y << endl;
}

int main() {
    node n;
    n.change_x(2);
    n.change_y(102);

    n.show_x();
    n.show_y();

    n.hello();
    my_friend(n);

    return 0;
}
```

#### 拷贝

```cpp
#include <iostream>
#include <cstdlib>
#include <cstring>
using namespace std;

class node {
public:
    int n;
    char *s;
    node() {
        this->n = 5;
        this->s = (char *)calloc(5, 1);
        s[0] = 'N';
    }
    node(int n, char *s) {
        this->n = n;
        this->s = (char *)calloc(strlen(s) + 1, 1);
        memcpy(this->s, s, strlen(s));
    }
    node(const node &a) {
        this->n = a.n;
        // 深拷贝
        this->s = (char *)calloc(strlen(a.s) + 1, sizeof(char));
        memcpy(this->s, a.s, strlen(a.s));

        // 浅拷贝
        this->s = a.s;
    }
    ~node() {
        cout << "d" << endl;
        free(this->s);
    }
};

int main() {
    node t;
    t.n = 10;
    // 内存泄漏
    t.s = (char *)calloc(8, 1);
    t.s[0] = 'A', t.s[1] = 'B', t.s[2] = 'a';
    cout << t.s << endl;

    node tt(t);
    tt.n = 20;
    cout << tt.n << " " << tt.s << endl;
    tt.s[0] = 'B';
    cout << tt.s << endl;
    cout << t.s << endl;

    return 0;
}
```



### 运算符重载

```cpp
#include <iostream>
using namespace std;

class node {
    public:
        int x, y;
        node() : x(0), y(0) {}
        node(int a, int b) : x(a), y(b) {}
        node(const node &a) : x(a.x), y(a.y) {}

        node operator-(const node &a) {
            node t;
            t.x -= a.x ;
            t.y -= a.y ;
            return t;
        }
};

node operator+(const node &a, const node &b) {
    node t;
    t.x = a.x + b.x;
    t.y = a.y + b.y;
    return t;
}

int main() {
    node a(3, 6);
    node b(4, 2);
    node c = a + b;
    node d = a - b;

    cout << c.x << " " << c.y << endl;
    cout << d.x << " " << d.y << endl;

    return 0;
}
```



涉及到比较排序时需要重载小于号

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

class node {
    public:
        int x, y;

        bool operator<(const node &b) const{
            if (this->x == b.x) {
                return this->y < b.y;
            }
            return this->x < b.x;
        }
};


int main() {
    int n;
    node num[105];
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> num[i].x >> num[i].y;
    }

    sort(num, num + n);
    for (int i = 0; i < n; i++) {
        cout << "[" << num[i].x << ", " << num[i].y << "]" << endl;
    }

    return 0;
}
```

### my_string

```cpp
#include <iostream>
#include <cstring>
using namespace std;

class my_string
{
private:
    int len;
    char *p;

public:
    my_string() : len(0), p(NULL) {}
    my_string(const char c[]){
        my_string(strlen(c), c);
    } 
    my_string(int len, const char s[])
    {
        if (this->len)
        {
            free(this->p);
        }
        this->len = len;
        this->p = (char *)calloc(strlen(s) + 1, 1);
        memcpy(this->p, s, strlen(s));
    }
    my_string(const my_string &b)
    {
        my_string(b.len, b.p);
    }

    friend istream &operator>>(istream &in, my_string &a);
    friend istream &operator<<(istream &in, my_string &a);
    friend ostream &operator<<(ostream &out, my_string &a);
    friend ostream &operator>>(ostream &out, my_string &a);
};

istream &operator>>(istream &in, my_string &a)
{
    if (a.len != 0)
    {
        free(a.p);
    }
    a.p = (char *)calloc(1005, 1);
    in >> a.p;
    a.len = strlen(a.p);
    return in;
};

istream &operator<<(istream &in, my_string &a)
{
    if (a.len != 0)
    {
        free(a.p);
    }
    a.p = (char *)calloc(1005, 1);
    in >> a.p;
    a.len = strlen(a.p);
    return in;
};

ostream &operator<<(ostream &out, my_string &a)
{
    if (a.len != 0)
    {
        out << a.p;
    }
    return out;
}

ostream &operator>>(ostream &out, my_string &a)
{
    if (a.len != 0)
    {
        out << a.p;
    }
    return out;
}

int main()
{
    my_string s = "abc";

    cout << s << endl;

}
```
