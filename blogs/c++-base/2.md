### priority_queue 优先队列（堆）

> 用数组模拟的完全二叉树
>
> 加入 删除 O(logn)
>
> \<queue>
>
> que.size(); O(1)
>
> que.push(x); O(logn)
>
> que.pop(); O(logn) // 堆顶
>
> que.top(); O(n)
>
> que.empty(); O(n)



```cpp
#include <iostream>
#include <queue>

using namespace std;

struct node
{
    int a, b;

    bool operator<(const node &b) const {
        if (this->a == b.a) {
            return this->b > b.b;
        }
        return this->a > b.a;
    }
};

struct cmp {
    bool operator() (const node &a, const node &b) {
        if (a.a == b.a) {
            return a.b > b.b;
        }
        return a.a > b.a;
    }
};

int main() {
    // priority_queue<node> que;
    priority_queue<node, vector<node>, cmp >que;
    que.push((node){2, 12});
    que.push((node){2, 1});
    que.push((node){99, 7});
    que.push((node){12, 7});
    que.push((node){-1, 0});

    while (!que.empty()) {
        cout << que.top().a << " " << que.top().b << endl;
        que.pop();
    }

    // priority_queue<int> que;

    // priority_queue<int, vector<int>, less<int> > que;

}
```

## 仿函数

```cpp
#include <iostream>

using namespace std;

struct my_less {
    bool operator() (const int &a, const int &b) {
        return a < b;
    }
};

struct my_add {
    int operator()(const int &a, const int &b) {
        return a + b;
    }
};

int main() {
    int a, b;
    cin >> a >> b;
    if (my_less{}(a, b)){
        cout << "a lass than b" << endl;
    } else {
        cout << "b less than a" << endl;
    }

    cout << my_add{}(a, b) << endl;

    return 0;
}
```
